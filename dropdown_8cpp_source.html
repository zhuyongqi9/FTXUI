29 /// @brief A dropdown menu.
30 /// @ingroup component
31 /// @param option The options for the dropdown.
32 // NOLINTNEXTLINE
33 Component Dropdown(DropdownOption option) {
34  class Impl : public ComponentBase, public DropdownOption {
35  public:
36  explicit Impl(DropdownOption option) : DropdownOption(std::move(option)) {
37  FillDefault();
38  checkbox_ = Checkbox(checkbox);
39  radiobox_ = Radiobox(radiobox);
40 
41  Add(Container::Vertical({
42  checkbox_,
43  Maybe(radiobox_, checkbox.checked),
44  }));
45  }
46 
47  Element Render() override {
48  radiobox.selected =
49  util::clamp(radiobox.selected(), 0, int(radiobox.entries.size()) - 1);
50  title_ = radiobox.entries[selected_()];
51 
52  return transform(*open_, checkbox_->Render(), radiobox_->Render());
53  }
54 
55  // Switch focus in between the checkbox and the radiobox when selecting it.
56  bool OnEvent(ftxui::Event event) override {
57  const bool open_old = open_();
58  const int selected_old = selected_();
59  bool handled = ComponentBase::OnEvent(event);
60 
61  // Transfer focus to the radiobox when the dropdown is opened.
62  if (!open_old && open_()) {
63  radiobox_->TakeFocus();
64  }
65 
66  // Auto-close the dropdown when the user selects an item, even if the item
67  // it the same as the previous one.
68  if (open_old && open_()) {
69  const bool should_close = (selected_() != selected_old) || //
70  (event == Event::Return) || //
71  (event == Event::Character(' ')) || //
72  (event == Event::Escape); //
73 
74  if (should_close) {
75  checkbox_->TakeFocus();
76  open_ = false;
77  handled = true;
78  }
79  }
80 
81  return handled;
82  }
83 
84  void FillDefault() {
85  open_ = checkbox.checked;
86  selected_ = radiobox.selected;
87  checkbox.checked = &*open_;
88  radiobox.selected = &*selected_;
89  checkbox.label = &title_;
90 
91  if (!checkbox.transform) {
92  checkbox.transform = [](const EntryState& s) {
93  auto prefix = text(s.state ? "↓ " : "→ "); // NOLINT
94  auto t = text(s.label);
95  if (s.active) {
96  t |= bold;
97  }
98  if (s.focused) {
99  t |= inverted;
100  }
101  return hbox({prefix, t});
102  };
103  }
104 
105  if (!transform) {
106  transform = [](bool is_open, Element checkbox_element,
107  Element radiobox_element) {
108  if (is_open) {
109  const int max_height = 12;
110  return vbox({
111  std::move(checkbox_element),
112  separator(),
113  std::move(radiobox_element) | vscroll_indicator | frame |
114  size(HEIGHT, LESS_THAN, max_height),
115  }) |
116  border;
117  }
118  return vbox({std::move(checkbox_element), filler()}) | border;
119  };
120  }
121  }
122 
123  private:
124  Ref<bool> open_;
125  Ref<int> selected_;
126  Component checkbox_;
127  Component radiobox_;
128  std::string title_;
129  };
130 
131  return Make<Impl>(option);
132 }
