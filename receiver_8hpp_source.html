80  bool Receive(T* t) {
81  while (senders_ || !queue_.empty()) {
82  std::unique_lock<std::mutex> lock(mutex_);
83  if (queue_.empty()) {
84  notifier_.wait(lock);
85  }
86  if (queue_.empty()) {
87  continue;
88  }
89  *t = std::move(queue_.front());
90  queue_.pop();
91  return true;
92  }
93  return false;
94  }
